#version 330 

	layout(location = 0) in vec4 position;
	uniform vec4 uniform_color;
	uniform mat4 m_view;
	uniform mat4 m_model;
	uniform mat4 m_projection;
	out vec4 theColor;
	void main()
	{
		mat4 m = m_projection * m_view * m_model;

		m[0][0] = 1;
		m[0][1] = 0;
		m[0][2] = 0;
		m[1][0] = 0;
		m[1][1] = 1;
		m[1][2] = 0;
		m[2][0] = 0;
		m[2][1] = 0;
		m[2][2] = 1;

		gl_Position = m * position;
		theColor = uniform_color;
	}
